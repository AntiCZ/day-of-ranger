Day of ranger - pure javascript calendar

date picker, range picker, multi picker with no dependencies and mobile friendly

Plugin is in alfa. Do not use for a productio yet.

Single pick day

Attribute "readonly" is required. Prevents show up a keyboard on mobile devices.

If you want allow writing in to the input follow example <link>.

    
    <input id="calendarSingleV1" type="text" readonly />
    
    
    (function(){
        var parseDate = function(date){ 
            return date.toISOString().split('T')[0];
        };
        var setValueToInput = function(date){ 
            document.querySelector('#calendarSingleV1').value = parseDate(date);
        };
        var calendarSingle = DORFactory.createCalendar('#calendarSingleV1', {
            afterPickDate: setValueToInput
        });
    })();

Attribute "readonly" is required. Prevents show up a keyboard on mobile devices.

If you want allow writing in to the input follow example <link>.

    
    <input id="calendarSingleV1" type="text" readonly />
    

    
    (function(){

            document.querySelector('#calendarPreselect').value = parseToISO(date);
        };
        var calendarSingle = DORFactory.createCalendar('#calendarPreselect', {
            datePreselected: new Date(), //if omit datePreselected and set afterInit you will get 1. 1. 1970

Single pick day, local date

Attribute "readonly" is required. Prevents show up a keyboard on mobile devices. If you want allow writing in to the input follow example <link>.

Why two inputs? Inspirate by input <input type="date">. Show to user his local date and for us programeres standardized ISO date. toLocaleDateString() accepts {Object} options for formating output follow links below.

Learn more about .toLocaleDateString() 

    
    <input id="calendarSingleV2" type="text" readonly />
    <input id="calendarSingleV2Hidden" type="hidden" />
    
    
    (function(){
        var setValueToInput = function(date, parseToISO, event) { 
            document.querySelector('#calendarSingleV2').value = date.toLocaleDateString();
            document.querySelector('#calendarSingleV2Hidden').value = parseToISO(date);
        };

        var calendarSingle = DORFactory.createCalendar('#calendarSingleV2', {
            afterPickDate: setValueToInput
        });
    })();

Range pick days - special Sundays and disabled past

    
    <input id="calendarRange" type="text" readonly />
    <input id="calendarRangeHiddenFrom" type="hidden" />
    <input id="calendarRangeHiddenTo" type="hidden" />
    
    
    (function(){
        var setValueToInput = function(date, parseToISO, event) { 
            var from = date.from.toLocaleDateString();
            var to = date.to.toLocaleDateString();
            document.querySelector('#calendarRange').value = from + ' \u2794 ' + to;
            document.querySelector('#calendarRangeHiddenFrom').value = parseToISO(date.from);
            document.querySelector('#calendarRangeHiddenTo').value = parseToISO(date.to);
        };
        var isDateSpecial = function(date) {
            return  date.getDay() === 0;
        };
        var isDateDisabled = function(date) {
            return  date < new Date().setHours(0,0,0,0);//For now, must be reset to midnight If Today is allowed
        };
        var calendarRange = DORFactory.createCalendar('#calendarRange', {
            type: 'range',
            afterPickDate: setValueToInput,
            isDateSpecial: isDateSpecial,
            isDateDisabled: isDateDisabled,
        });
    })();

Range pick days - Close after pick dates


<input id="calendarRange" type="text" readonly />
<input id="calendarRangeHiddenFrom" type="hidden" />
<input id="calendarRangeHiddenTo" type="hidden" />


(function(){
    var setValueToInput = function(date, parseToISO, event) { 
        var from = date.from.toLocaleDateString();
        var to = date.to.toLocaleDateString();
        document.querySelector('#calendarRangeClose').value = from + ' \u2794 ' + to;
        document.querySelector('#calendarRangeCloseHiddenFrom').value = parseToISO(date.from);
        document.querySelector('#calendarRangeCloseHiddenTo').value = parseToISO(date.to);
        setTimeout(function(){ 
            date.from < date.to ? this.hide() : false;
        }.bind(this), 300);
    };
    var calendarRange = DORFactory.createCalendar('#calendarRangeClose', {
        type: 'range',
        afterPickDate: setValueToInput,
    });
})();
